import prismaChat from '../../../chat-processor/src/prisma/client';
import { AppLogger } from '../config/logger';
 
export interface ConversationVerificationResult {
  conversationId: string;
  externalId: string;
  steps: {
    imported: boolean;
    analysisCreated: boolean;
    analysisCompleted: boolean;
    messagesSent: boolean;
    mediaProcessed?: boolean;
  };
  issues: string[];
  status: 'complete' | 'incomplete' | 'failed';
  lastProcessedAt?: Date;
}
 
export interface VerificationSummary {
  totalConversations: number;
  completeConversations: number;
  incompleteConversations: number;
  failedConversations: number;
  issueBreakdown: Record<string, number>;
  details: ConversationVerificationResult[];
}
 
export class ConversationVerificationService {
  /**
   * Verifica todas las conversaciones de una instancia para garantizar
   * que cada una haya completado todos los pasos necesarios
   */
  async verifyInstanceConversations(
    instanceId: number,
    options: {
      limit?: number;
      offset?: number;
      includeCompleted?: boolean;
      sinceDays?: number;
    } = {}
  ): Promise<VerificationSummary> {
    AppLogger.info('🔍 Starting comprehensive conversation verification', {
      instanceId,
      options
    });
 
    // Obtener configuración de la instancia
    const config = await prismaChat.chatProcessorInstance.findUnique({
      where: { instanceId }
    });
 
    if (!config) {
      throw new Error(`Chat Processor configuration not found for instance ${instanceId}`);
    }
 
    // Construir filtros de fecha si se especifican
    const dateFilter = options.sinceDays ? {
      importedAt: {
        gte: new Date(Date.now() - (options.sinceDays * 24 * 60 * 60 * 1000))
      }
    } : {};
 
    // Obtener todas las conversaciones con sus análisis
    const conversations = await prismaChat.importedConversation.findMany({
      where: {
        instanceId: config.id,
        ...dateFilter
      },
      include: {
        analysis: true
      },
      orderBy: { importedAt: 'desc' },
      take: options.limit || 1000,
      skip: options.offset || 0
    });
 
    AppLogger.info('📊 Retrieved conversations for verification', {
      count: conversations.length,
      instanceId
    });
 
    // Verificar cada conversación individual
    const verificationResults: ConversationVerificationResult[] = [];
    const issueBreakdown: Record<string, number> = {};
 
    for (const conversation of conversations) {
      const result = await this.verifySingleConversation(conversation);
      
      // Solo incluir si hay problemas o si se solicitan las completas
      if (result.status !== 'complete' || options.includeCompleted) {
        verificationResults.push(result);
      }
 
      // Acumular estadísticas de problemas
      result.issues.forEach(issue => {
        issueBreakdown[issue] = (issueBreakdown[issue] || 0) + 1;
      });
    }
 
    const summary: VerificationSummary = {
      totalConversations: conversations.length,
      completeConversations: verificationResults.filter(r => r.status === 'complete').length,
      incompleteConversations: verificationResults.filter(r => r.status === 'incomplete').length,
      failedConversations: verificationResults.filter(r => r.status === 'failed').length,
      issueBreakdown,
      details: verificationResults
    };
 
    AppLogger.info('✅ Conversation verification completed', {
      instanceId,
      summary: {
        total: summary.totalConversations,
        complete: summary.completeConversations,
        incomplete: summary.incompleteConversations,
        failed: summary.failedConversations,
        mainIssues: Object.entries(issueBreakdown)
          .sort(([,a], [,b]) => b - a)
          .slice(0, 3)
          .map(([issue, count]) => `${issue}: ${count}`)
      }
    });
 
    return summary;
  }
 
  /**
   * Verifica una conversación individual paso a paso
   */
  private async verifySingleConversation(
    conversation: any
  ): Promise<ConversationVerificationResult> {
    const issues: string[] = [];
    const steps = {
      imported: false,
      analysisCreated: false,
      analysisCompleted: false,
      messagesSent: false,
      mediaProcessed: false
    };
 
    // 1. PASO: IMPORTACIÓN
    if (conversation.id && conversation.externalId && conversation.content) {
      steps.imported = true;
    } else {
      issues.push('missing_import_data');
    }
 
    // 2. PASO: ANÁLISIS CREADO
    if (conversation.analysis) {
      steps.analysisCreated = true;
 
      // 3. PASO: ANÁLISIS COMPLETADO
      if (conversation.analysis.status === 'completed' && 
          conversation.analysis.analysisText && 
          conversation.analysis.responseReceivedAt) {
        steps.analysisCompleted = true;
      } else if (conversation.analysis.status === 'failed') {
        issues.push('analysis_failed');
      } else if (conversation.analysis.status === 'pending') {
        issues.push('analysis_pending');
      } else {
        issues.push('analysis_incomplete');
      }
    } else {
      issues.push('missing_analysis');
    }
 
    // 4. PASO: MENSAJES ENVIADOS
    // Verificar si se enviaron mensajes a agentes
    const messagesSent = await this.checkMessagesSent(conversation.externalId);
    if (messagesSent) {
      steps.messagesSent = true;
    } else {
      issues.push('messages_not_sent');
    }
 
    // 5. PASO: MULTIMEDIA PROCESADA (opcional)
    if (conversation.metadata?.hasMedia) {
      const mediaProcessed = await this.checkMediaProcessed(conversation.externalId);
      steps.mediaProcessed = mediaProcessed;
      if (!mediaProcessed) {
        issues.push('media_not_processed');
      }
    } else {
      steps.mediaProcessed = true; // No hay multimedia que procesar
    }
 
    // 6. VERIFICAR ESTADO DE PROCESAMIENTO
    if (conversation.processingStatus !== 'completed' && issues.length === 0) {
      issues.push('status_not_updated');
    }
 
    // 7. VERIFICAR CONSISTENCIA TEMPORAL
    if (conversation.analysis?.responseReceivedAt && 
        !conversation.lastProcessedAt) {
      issues.push('timestamps_inconsistent');
    }
 
    // Determinar estado general
    let status: 'complete' | 'incomplete' | 'failed';
    if (issues.length === 0 && Object.values(steps).every(step => step === true)) {
      status = 'complete';
    } else if (issues.some(issue => issue.includes('failed'))) {
      status = 'failed';
    } else {
      status = 'incomplete';
    }
 
    return {
      conversationId: conversation.id.toString(),
      externalId: conversation.externalId,
      steps,
      issues,
      status,
      lastProcessedAt: conversation.lastProcessedAt
    };
  }
 
  /**
   * Verifica si se enviaron mensajes para una conversación específica
   */
  private async checkMessagesSent(externalId: string): Promise<boolean> {
    try {
      // Buscar en logs de procesamiento si se generaron mensajes para esta conversación
      const recentLogs = await prismaChat.processingLog.findMany({
        where: {
          metadata: {
            path: ['conversationsProcessed'],
            array_contains: externalId
          },
          messagesGenerated: {
            gt: 0
          }
        },
        orderBy: { startedAt: 'desc' },
        take: 5
      });
 
      return recentLogs.length > 0;
    } catch (error) {
      AppLogger.error('Error checking messages sent', { externalId, error });
      return false;
    }
  }
 
  /**
   * Verifica si se procesó multimedia para una conversación específica
   */
  private async checkMediaProcessed(externalId: string): Promise<boolean> {
    try {
      // Buscar en logs si hay evidencia de procesamiento de multimedia
      const mediaLogs = await prismaChat.processingLog.findMany({
        where: {
          metadata: {
            path: ['mediaProcessingResults'],
            array_contains: [{ conversationId: externalId }]
          }
        },
        orderBy: { startedAt: 'desc' },
        take: 1
      });
 
      return mediaLogs.length > 0;
    } catch (error) {
      AppLogger.error('Error checking media processed', { externalId, error });
      return true; // Asumimos que está OK si no podemos verificar
    }
  }
 
  /**
   * Reprocesa conversaciones que no completaron todos los pasos
   */
  async reprocessIncompleteConversations(
    instanceId: number,
    options: {
      maxConversations?: number;
      issueTypes?: string[];
      dryRun?: boolean;
    } = {}
  ): Promise<{ reprocessed: number; errors: string[] }> {
    AppLogger.info('🔧 Starting reprocessing of incomplete conversations', {
      instanceId,
      options
    });
 
    const verification = await this.verifyInstanceConversations(instanceId, {
      includeCompleted: false,
      limit: options.maxConversations || 100
    });
 
    let conversationsToReprocess = verification.details.filter(
      detail => detail.status === 'incomplete'
    );
 
    // Filtrar por tipos de problemas si se especifican
    if (options.issueTypes && options.issueTypes.length > 0) {
      conversationsToReprocess = conversationsToReprocess.filter(
        detail => detail.issues.some(issue => options.issueTypes!.includes(issue))
      );
    }
 
    const errors: string[] = [];
    let reprocessedCount = 0;
 
    if (options.dryRun) {
      AppLogger.info('🧪 DRY RUN: Would reprocess conversations', {
        count: conversationsToReprocess.length,
        conversations: conversationsToReprocess.map(c => ({
          externalId: c.externalId,
          issues: c.issues
        }))
      });
      return { reprocessed: conversationsToReprocess.length, errors: [] };
    }
 
    // Marcar conversaciones para reprocesamiento
    for (const detail of conversationsToReprocess) {
      try {
        await prisma.importedConversation.update({
          where: {
            id: parseInt(detail.conversationId)
          },
          data: {
            processingStatus: 'pending',
            lastProcessedAt: null
          }
        });
 
        // Si falta análisis, recrear registro de análisis
        if (detail.issues.includes('missing_analysis') || 
            detail.issues.includes('analysis_failed')) {
          await this.recreateAnalysisRecord(parseInt(detail.conversationId));
        }
 
        reprocessedCount++;
        AppLogger.debug('✅ Marked conversation for reprocessing', {
          externalId: detail.externalId,
          issues: detail.issues
        });
 
      } catch (error: any) {
        const errorMsg = `Failed to reprocess ${detail.externalId}: ${error.message}`;
        errors.push(errorMsg);
        AppLogger.error('❌ Error marking conversation for reprocessing', {
          externalId: detail.externalId,
          error: error.message
        });
      }
    }
 
    AppLogger.info('✅ Reprocessing setup completed', {
      instanceId,
      reprocessed: reprocessedCount,
      errors: errors.length,
      total: conversationsToReprocess.length
    });
 
    return { reprocessed: reprocessedCount, errors };
  }
 
  /**
   * Recrea un registro de análisis para una conversación
   */
  private async recreateAnalysisRecord(conversationId: number): Promise<void> {
    try {
      // Obtener la conversación y configuración
      const conversation = await prismaChat.importedConversation.findUnique({
        where: { id: conversationId },
        include: {
          chatProcessorInstance: true,
          analysis: true
        }
      });
 
      if (!conversation) {
        throw new Error(`Conversation ${conversationId} not found`);
      }
 
      // Si ya existe análisis, actualizar su estado
      if (conversation.analysis) {
        await prismaChat.conversationAnalysis.update({
          where: { id: conversation.analysis.id },
          data: {
            status: 'pending',
            retryCount: conversation.analysis.retryCount + 1,
            errorMessage: null,
            requestSentAt: null,
            responseReceivedAt: null
          }
        });
      } else {
        // Crear nuevo registro de análisis
        const analysisConfig = (conversation.chatProcessorInstance.configuration as any)?.analysisConfig as any;
        await prismaChat.conversationAnalysis.create({
          data: {
            conversationId: conversation.id,
            analysisFlowId: analysisConfig?.flowId || 'unknown',
            status: 'pending'
          }
        });
      }
 
      AppLogger.debug('✅ Analysis record recreated', {
        conversationId,
        externalId: conversation.externalId
      });
 
    } catch (error: any) {
      AppLogger.error('❌ Error recreating analysis record', {
        conversationId,
        error: error.message
      });
      throw error;
    }
  }
 
  /**
   * Obtiene estadísticas detalladas de verificación
   */
  async getVerificationStats(instanceId: number): Promise<{
    overview: {
      totalConversations: number;
      completedPercentage: number;
      avgCompletionTime: number;
      oldestIncomplete: Date | null;
    };
    stepStats: {
      imported: number;
      analysisCreated: number;
      analysisCompleted: number;
      messagesSent: number;
      mediaProcessed: number;
    };
    issueFrequency: Array<{ issue: string; count: number; percentage: number }>;
  }> {
    const verification = await this.verifyInstanceConversations(instanceId, {
      includeCompleted: true,
      limit: 5000 // Muestra amplia para estadísticas
    });
 
    // Calcular estadísticas de pasos
    const stepStats = {
      imported: 0,
      analysisCreated: 0,
      analysisCompleted: 0,
      messagesSent: 0,
      mediaProcessed: 0
    };
 
    verification.details.forEach(detail => {
      if (detail.steps.imported) stepStats.imported++;
      if (detail.steps.analysisCreated) stepStats.analysisCreated++;
      if (detail.steps.analysisCompleted) stepStats.analysisCompleted++;
      if (detail.steps.messagesSent) stepStats.messagesSent++;
      if (detail.steps.mediaProcessed) stepStats.mediaProcessed++;
    });
 
    // Calcular frecuencia de problemas
    const issueFrequency = Object.entries(verification.issueBreakdown)
      .map(([issue, count]) => ({
        issue,
        count,
        percentage: Math.round((count / verification.totalConversations) * 100)
      }))
      .sort((a, b) => b.count - a.count);
 
    // Encontrar la conversación incompleta más antigua
    const incompleteConversations = verification.details.filter(d => d.status === 'incomplete');
    const oldestIncomplete = incompleteConversations.length > 0 
      ? incompleteConversations
          .filter(d => d.lastProcessedAt)
          .sort((a, b) => a.lastProcessedAt!.getTime() - b.lastProcessedAt!.getTime())[0]?.lastProcessedAt || null
      : null;
 
    return {
      overview: {
        totalConversations: verification.totalConversations,
        completedPercentage: Math.round((verification.completeConversations / verification.totalConversations) * 100),
        avgCompletionTime: 0, // Calcular promedio de tiempo de procesamiento
        oldestIncomplete
      },
      stepStats,
      issueFrequency
    };
  }
